30      @Test
31      public void testGenerateMappers1() throws Exception {
32          MapperClass mapperClass = new MapperClass("at.rseiler.spbee.test.mapper.TestMapper", "at.rseiler.spbee.test.entity.TestEntity");
33          String javaCode = generateJavaCode(mapperClass);
34  
35          assertContains(javaCode,
36                  "package at.rseiler.spbee.test.mapper;",
37                  "import java.sql.ResultSet;",
38                  "import javax.annotation.Generated;",
39                  "import at.rseiler.spbee.test.entity.TestEntity;",
40                  "import org.springframework.jdbc.core.RowMapper;",
41                  "@Generated(value = \"at.rseiler.spbee.core.SPBeeAnnotationProcessor\", date = ",
42                  "public class TestMapper",
43                  "implements RowMapper<TestEntity>",
44                  "public TestEntity mapRow(ResultSet rs, int rowNum)",
45                  "return new TestEntity();"
46          );
47      }

49      @Test
50      public void testGenerateMappers2() throws Exception {
51          MapperClass mapperClass = new MapperClass("at.rseiler.spbee.test.mapper.TestMapper", "at.rseiler.spbee.test.entity.TestEntity");
52          mapperClass.addVariable(new Variable("id", int.class.getCanonicalName()));
53          mapperClass.addVariable(new Variable("name", String.class.getCanonicalName()));
54          mapperClass.addVariable(new Variable("isTest", Boolean.class.getCanonicalName()));
55  
56          String javaCode = generateJavaCode(mapperClass);
57  
58          assertContains(javaCode,
59                  "return new TestEntity(rs.getInt(1), rs.getString(2), rs.getBoolean(3));"
60          );
61      }
62  
63      private String generateJavaCode(MapperClass mapperClass) throws IOException, JClassAlreadyExistsException, ClassNotFoundException {
64          Filer filer = mock(Filer.class);
65          JavaFileObject javaFileObject = mock(JavaFileObject.class);
66          StringWriter stringWriter = new StringWriter();
67  
68          when(processingEnv.getFiler()).thenReturn(filer);
69          when(filer.createSourceFile("at.rseiler.spbee.test.mapper.TestMapper")).thenReturn(javaFileObject);
70          when(javaFileObject.openWriter()).thenReturn(stringWriter);
71  
72          MapperGenerator mapperGenerator = new MapperGenerator(processingEnv);
73          List<MapperClass> mapperClasses = Arrays.asList(mapperClass);
74          mapperGenerator.generateMappers(mapperClasses);
75  
76          return stringWriter.toString();
77      }
