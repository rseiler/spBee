package at.rseiler.spbee.core.pojo;

import java.util.Optional;

/**
 * Holds the type information which consists of the type and the generic type.
 *
 * @author Reinhard Seiler {@literal <reinhard.seiler@gmail.com>}
 */
public class TypeInfo {

    private final String type;
    private final Optional<String> genericType;

    /**
     * Constructs a new TypeInfo so that it represents a basic type.
     *
     * @param type the type
     */
    public TypeInfo(String type) {
        this(type, Optional.empty());
    }

    /**
     * Constructs a new TypeInfo so that it represents a type with generics.
     *
     * @param type        the type
     * @param genericType the generic type
     */
    public TypeInfo(String type, String genericType) {
        this(type, Optional.of(genericType));
    }

    /**
     * Constructs a new TypeInfo so that it represents a basic type or a type with generics.
     *
     * @param type        the type
     * @param genericType the generic type
     */
    public TypeInfo(String type, Optional<String> genericType) {
        this.type = type;
        this.genericType = genericType;
    }

    /**
     * Returns the base type.
     * E.g.: String -> String; List<String> -> List
     *
     * @return the base type
     */
    public String getType() {
        return type;
    }

    /**
     * Returns the generic type.
     * E.g.: List<String> -> String; String -> Optional.empty()
     *
     * @return the generic type
     */
    public Optional<String> getGenericType() {
        return genericType;
    }

    /**
     * Returns the generic type if it exists otherwise the base type.
     * E.g.: String -> String; List<String> -> String
     *
     * @return the generic type or the base type
     */
    public String getGenericTypeOrType() {
        return genericType.orElse(type);
    }

    /**
     * Returns the string representation of the type and generic type.
     *
     * @return the string representation
     */
    public String asString() {
        return genericType.isPresent() ? type + '<' + genericType + '>' : type;
    }

    @Override
    public String toString() {
        return "TypeInfo{" +
                "type='" + type + '\'' +
                ", genericType=" + genericType +
                '}';
    }

}
